My PyTorch model is overfitting quickly, how can I implement effective regularization?

python 
import torch.nn as nn

class RegularizedModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.dropout = nn.Dropout(p=0.5)
        self.bn = nn.BatchNorm1d(256)

    def forward(self, x):
        x = self.bn(self.dropout(x))
        return x
